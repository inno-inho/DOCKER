# 현재 도커 이미지 목록 확인
docker images

# 이미지 삭제
docker rmi[id or 이미지명]

# 도커 이미지 내 진입
docker run -it [id or 이미지명] sh

# 임시컨테이너 생성(종료시 삭제) + 포트연결
# 3000 -> 외부 포트 번호, 80-> 도커 내부 포트번호, fn:1.1 -> image 이름 
docker run -it --rm -p 3000:80 fn:1.1


# tagging
docker tag bn innosama/inno:bn
docker tag fn innosama/inno:fn
docker tag redis innosama/inno:redis
docker tag db innosama/inno:db

# push
docker push innosama/inno:bn
docker push innosama/inno:fn
docker push innosama/inno:redis
docker push innosama/inno:db
